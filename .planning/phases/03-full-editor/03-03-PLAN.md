<objective>
Upgrade SidePanel from read-only (Phase 2) to full editor (EDIT-10, EDIT-11, EDIT-12). Add a CodeMirror 6 notes editor with Edit/Preview/Split segmented toggle, a metadata key-value table with add/edit/delete, and editable title + status + type fields that appear on demand.

- MetadataEditor with row add/edit/delete (D-12, EDIT-11)

</objective>

<design_note>
**Preview-default + [E] button pattern (D-10):** prevents accidental edits while browsing. Panel mounts in preview mode matching Phase 2 behavior; clicking any field value or the [E] icon on hover switches the whole panel to edit mode. Single `isEditing` boolean — all fields flip together. Exit via Escape (returns focus to canvas per D-07) or clicking outside the panel.

**CodeMirror lifecycle (research Pattern 4):** `useEffect` creates `EditorView` once per `nodeId` (remount when node switches). `EditorView.updateListener.of` calls a debounced callback that invokes `updateNodeNotes` after 1000ms of inactivity (D-11). Debounce lives in a `useRef<Timer>` so remounting clears stale timers.

**CodeMirror theme via CSS variables (research Pattern 4):** `EditorView.theme({...}, {dark: true})` with `var(--rv-*)` values. The `dark: true` flag only affects CodeMirror internal defaults — our explicit CSS-variable rules dominate and automatically invert when the app theme switches.

**Split view narrow-panel behavior (UI-SPEC):** when panel width < 560px, split mode collapses to Edit mode with a note "Panel too narrow for split view." (11px, `--rv-text-tertiary`). Detection via panel `width` state that already exists in SidePanel.tsx.

**Metadata editor (D-12):** grid layout with 3 columns (key input | value input | × button) + a "+ Add row" button below. Empty state copy: "No metadata. Click + to add a key-value pair." Each row edit triggers `updateNodeMetadata(nodeId, newMetadataRecord)` — in-place, no dataKey bump.
</design_note>

    ```typescript
    import { useEffect, useRef } from "react";
    import { EditorState } from "@codemirror/state";
    import { EditorView, drawSelection, highlightActiveLine, keymap, placeholder as cmPlaceholder } from "@codemirror/view";
    import { defaultKeymap, history, historyKeymap } from "@codemirror/commands";
    import { markdown, markdownLanguage } from "@codemirror/lang-markdown";
    import { bracketMatching, defaultHighlightStyle, indentOnInput, syntaxHighlighting } from "@codemirror/language";
    import { closeBrackets } from "@codemirror/autocomplete";
    import { codemirrorRvTheme } from "../theme/codemirrorTheme";

    interface Options {
      container: React.RefObject<HTMLDivElement>;
      nodeId: string;
      initialDoc: string;
      onPersist: (nodeId: string, content: string) => void;
      debounceMs?: number;
      placeholder?: string;
    }

    export function useCodeMirror({ container, nodeId, initialDoc, onPersist, debounceMs = 1000, placeholder: ph = "Write notes in markdown…" }: Options) {
      const viewRef = useRef<EditorView | null>(null);
      const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
      const onPersistRef = useRef(onPersist);
      onPersistRef.current = onPersist;

      useEffect(() => {
        if (!container.current) return;

        const extensions = [
          history(),
          drawSelection(),
          indentOnInput(),
          bracketMatching(),
          closeBrackets(),
          highlightActiveLine(),
          markdown({ base: markdownLanguage, codeLanguages: [] }),
          syntaxHighlighting(defaultHighlightStyle, { fallback: true }),
          keymap.of([...defaultKeymap, ...historyKeymap]),
          cmPlaceholder(ph),
          EditorView.lineWrapping,
          codemirrorRvTheme,
          EditorView.updateListener.of((update) => {
            if (!update.docChanged) return;
            if (timerRef.current) clearTimeout(timerRef.current);
            const content = update.state.doc.toString();
            timerRef.current = setTimeout(() => {
              onPersistRef.current(nodeId, content);
              timerRef.current = null;
            }, debounceMs);
          }),
        ];

        const view = new EditorView({
          state: EditorState.create({ doc: initialDoc, extensions }),
          parent: container.current,
        });
        viewRef.current = view;

        return () => {
          if (timerRef.current) { clearTimeout(timerRef.current); timerRef.current = null; }
          view.destroy();
          viewRef.current = null;
        };
      }, [nodeId, debounceMs, ph]);

      return viewRef;
    }
    ```

    The hook is stable for `onPersist` changes via `onPersistRef` (avoids remount on every React render). Remount ONLY when `nodeId` changes (user switched nodes).

    ```tsx
    import { useEffect, useState } from "react";

    interface MetadataEditorProps {
      metadata: Record<string, unknown>;
      onChange: (metadata: Record<string, unknown>) => void;
    }

    interface Row { key: string; value: string }

    function toRows(m: Record<string, unknown>): Row[] {
      return Object.entries(m).map(([key, value]) => ({
        key,
        value: typeof value === "string" ? value : JSON.stringify(value),
      }));
    }

    function toRecord(rows: Row[]): Record<string, unknown> {
      const out: Record<string, unknown> = {};
      for (const { key, value } of rows) {
        if (!key.trim()) continue;
        out[key] = value;
      }
      return out;
    }

    export function MetadataEditor({ metadata, onChange }: MetadataEditorProps) {
      const [rows, setRows] = useState<Row[]>(() => toRows(metadata));

      // Sync when the underlying metadata changes externally
      useEffect(() => { setRows(toRows(metadata)); }, [metadata]);

      const persist = (next: Row[]) => {
        setRows(next);
        onChange(toRecord(next));
      };

      if (rows.length === 0) {
        return (
          <div className="flex flex-col gap-2">
            <div className="text-[13px] text-rv-text-tertiary">No metadata. Click + to add a key-value pair.</div>
            <AddButton onClick={() => persist([{ key: "", value: "" }])} />
          </div>
        );
      }

      return (
        <div className="flex flex-col gap-2">
          <div className="grid grid-cols-[1fr_1fr_auto] gap-x-2 gap-y-1 items-center">
            {rows.map((row, idx) => (
              <RowControls
                key={`${idx}-${row.key}`}
                row={row}
                onKeyChange={(k) => persist(rows.map((r, i) => i === idx ? { ...r, key: k } : r))}
                onValueChange={(v) => persist(rows.map((r, i) => i === idx ? { ...r, value: v } : r))}
                onRemove={() => persist(rows.filter((_, i) => i !== idx))}
              />
    ```
