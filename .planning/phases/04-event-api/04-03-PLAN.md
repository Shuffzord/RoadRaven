```typescript
applyEventBatch: (updates) => {
  if (updates.length === 0) return;
  const state = get();
  const nodeIndex = state.nodeIndex;
  const nextLiveEventMeta = { ...state.liveEventMeta };
  // In-place mutation via nodeIndex (matches updateNodeStatus pattern — preserves the data reference, so react-d3-tree does NOT deep-clone, and dataKey stays stable)
  for (const u of updates) {
    const node = nodeIndex.get(u.nodeId);
    if (node && node.status !== u.status) {
      node.status = u.status as typeof node.status;
    }
    nextLiveEventMeta[u.nodeId] = {
      lastEventAt: u.lastEventAt,
      source: u.source,
      meta: u.meta,
    };
  }
  // Single set() call — Pitfall 6: exactly one bump of statusTick per batch
  set({
    liveEventMeta: nextLiveEventMeta,
    statusTick: state.statusTick + 1,
    // dataKey: state.dataKey, // INTENTIONALLY NOT BUMPED — Phase 4 overlay is status-only
  });
},
```

export function useIsNodeLive(nodeId: string): boolean {
  return useRoadmapStore((s) => {
    const meta = s.liveEventMeta[nodeId];
    // Read liveTick to subscribe; Date.now() outside set() is fine
    void s.liveTick;

}

import { create } from "zustand";

export type ToastType = "malformed" | "unknown_node" | "invalid_status" | "disconnect";

export interface ActiveToast {
  id: string;
  type: ToastType;
  source: string;
  detail?: string;
  count: number;
  lastEventAt: number;
}

export const THROTTLE_WINDOW_MS = 5000; // D-24
export const MAX_STACKED = 3;           // UI-SPEC

interface ToastState {
  toasts: ActiveToast[];
  pushToast: (incoming: { type: ToastType; source: string; detail?: string }) => void;
  dismissToast: (id: string) => void;
}

export const useToastStore = create<ToastState>((set) => ({
  toasts: [],
  pushToast: (incoming) => set((state) => {
    const now = Date.now();
    const existing = state.toasts.find(
      (t) => t.type === incoming.type && t.source === incoming.source && (now - t.lastEventAt) < THROTTLE_WINDOW_MS,
    );
    if (existing) {
      return {
        toasts: state.toasts.map((t) =>
          t.id === existing.id ? { ...t, count: t.count + 1, lastEventAt: now } : t,
        ),
      };
    }
    const next: ActiveToast = {
      id: crypto.randomUUID(),
      type: incoming.type,
      source: incoming.source,
      detail: incoming.detail,
      count: 1,
      lastEventAt: now,
    };
    const updated = [...state.toasts, next];
    return {
      toasts: updated.length > MAX_STACKED ? updated.slice(updated.length - MAX_STACKED) : updated,
    };
  }),
  dismissToast: (id) => set((state) => ({ toasts: state.toasts.filter((t) => t.id !== id) })),
}));
