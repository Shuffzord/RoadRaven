**B. Create MarkdownRenderer.tsx:**

Build the remark/rehype pipeline per RESEARCH.md Pattern 5 and UI-SPEC Markdown Rendering Contract:

```typescript
import { unified } from "unified";
import remarkParse from "remark-parse";
import remarkGfm from "remark-gfm";
import remarkRehype from "remark-rehype";
import rehypeReact from "rehype-react";
import { Fragment, jsx, jsxs } from "react/jsx-runtime";

const processor = unified()
  .use(remarkParse)
  .use(remarkGfm)
  .use(remarkRehype)
  .use(rehypeReact, {
    Fragment,
    jsx,
    jsxs,
    components: {
      // Per UI-SPEC Markdown Rendering Contract:
      h1: (props) => <h1 className="text-[14px] font-semibold leading-[1.3] text-rv-text-primary mb-4" {...props} />,
      h2: (props) => <h2 className="text-[14px] font-semibold leading-[1.3] text-rv-text-primary mb-2" {...props} />,
      h3: (props) => <h3 className="text-[13px] font-semibold leading-[1.3] text-rv-text-primary mb-2" {...props} />,
      p: (props) => <p className="text-[13px] leading-[1.7] text-rv-text-secondary mb-2" {...props} />,
      a: ({ href, children, ...rest }) => <a href={href} className="text-rv-accent no-underline hover:underline" {...rest}>{children}</a>,
      code: ({ children, className, ...rest }) => {
        // Detect block code (has className with language-*) vs inline code
        if (className) {
          return <code className={`text-[11px] font-[Space_Grotesk] ${className}`} {...rest}>{children}</code>;
        }
        return <code className="text-[11px] font-[Space_Grotesk] bg-rv-bg-elevated text-rv-accent rounded-[3px] px-1 py-0.5" {...rest}>{children}</code>;
      },
      pre: ({ children, ...rest }) => <pre className="text-[11px] font-[Space_Grotesk] bg-rv-bg-elevated border border-rv-border rounded-[6px] p-2 mb-2 overflow-x-auto" {...rest}>{children}</pre>,
      ul: (props) => <ul className="ml-4 mb-2 list-disc" {...props} />,
      ol: (props) => <ol className="ml-4 mb-2 list-decimal" {...props} />,
      li: (props) => <li className="text-[13px] leading-[1.7] text-rv-text-secondary mb-1" {...props} />,
      blockquote: (props) => <blockquote className="border-l-[3px] border-rv-accent pl-4 text-rv-text-secondary italic mb-2" {...props} />,
      table: (props) => <table className="w-full border-collapse mb-2" {...props} />,
      th: (props) => <th className="text-[11px] font-semibold uppercase bg-rv-bg-elevated p-2 text-left border-b border-rv-border-subtle" {...props} />,
      td: (props) => <td className="text-[12px] p-2 border-b border-rv-border-subtle" {...props} />,
      del: (props) => <del className="text-rv-text-tertiary" {...props} />,
      input: ({ type, checked, ...rest }) => {
        if (type === "checkbox") {
          return <span className={`inline-block w-[14px] h-[14px] border border-rv-border rounded-[2px] mr-1.5 align-middle ${checked ? "bg-rv-accent" : ""}`} aria-checked={!!checked} role="checkbox" />;
        }
        return <input type={type} {...rest} />;
      },
    }
  });

export function MarkdownRenderer({ content }: { content: string }) {
  const rendered = useMemo(() => {
    const file = processor.processSync(content);
    return file.result as React.ReactNode;
  }, [content]);

  return <div className="markdown-notes">{rendered}</div>;
}
```
