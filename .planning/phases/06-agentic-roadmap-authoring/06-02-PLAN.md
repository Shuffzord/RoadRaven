```typescript
import type { ServerWebSocket } from "bun";
import { getLogger } from "@logtape/logtape";
import type { RoadmapRPCType } from "../../../../shared/types";
import type { AgentRequest } from "./eventSchema";
import { getOwnership } from "./refMap";
import {
  flushPending,
  isPathWithinMainDir,
  pushDialogAllowlistPath,
} from "./saveFile";
import { loadSettings } from "./settings";

const agentLogger = getLogger(["roadraven", "agent"]);

type WsData = { id: string; source?: string };

type RpcAgentRequest = (params: {
  tool: string;
  args: Record<string, unknown>;
}) => Promise<
  | { ok: true; data: unknown }
  | { ok: false; error: string; code: string; hint?: string; data?: unknown }
>;

interface MainWindowLike {
  webview: { rpc?: { request: { agentRequest: RpcAgentRequest } } };
}

interface ErrorPayload {
  code: string;
  message: string;
  hint?: string;
  data?: unknown;
}

/**
 * Phase 6 PLUG-AGENT-TRANSPORT-01: Bun-side request router. Receives every
 * `{type:'request'}` frame from eventServer's message dispatcher BEFORE the
 * EventCoalescer (Phase 4 D-25) is engaged. Gates kill-switch (D-15),
 * cross-ref boundary (RESEARCH L-08 Option B / EDIT-16), and path-allowlist
 * (D-13) BEFORE forwarding to the renderer's agentRequest RPC. Cycle
 * detection (moveNode descendant check) is enforced by the renderer
 * dispatcher because it owns the schema; this handler trusts the structured
 * error code that comes back.
 */
export async function agentRequestHandler(
  ws: ServerWebSocket<WsData>,
  request: AgentRequest,
  mainWindow: MainWindowLike,
): Promise<void> {
  const start = Date.now();
  const { id, method: tool, params: args } = request;

  // Gate 1 — Kill-switch (D-15 / SAFETY-03)
  const settings = loadSettings();
  if (settings.agentApi?.enabled === false) {
    sendError(ws, id, {
      code: "agent_api_disabled",
      message: "Agent API is disabled in application settings.",
      hint: "Enable it in .roadmap-settings.json: agentApi.enabled = true",
    });
    agentLogger.warn`Agent kill-switch blocked tool=${tool}`;
    return;
  }

  // Gate 2 — Cross-ref ownership boundary (EDIT-16 / RESEARCH L-08 Option B)
  if (tool === "moveNode") {
    const ownership = getOwnership();
    if (ownership.size > 0) {
      const nodeId = args.nodeId as string | undefined;
      const newParentId = args.newParentId as string | undefined;
      const nodeOwner = nodeId ? ownership.get(nodeId) : undefined;
      const parentOwner = newParentId ? ownership.get(newParentId) : undefined;
      if (nodeOwner && parentOwner && nodeOwner !== parentOwner) {
        sendError(ws, id, {
          code: "cross_ref_boundary",
          message: "Cannot move node across $ref file boundaries.",
          hint: "Phase 3 EDIT-16: cross-boundary moves are blocked.",
        });
        agentLogger.warn`moveNode cross-ref blocked nodeId=${nodeId} newParentId=${newParentId}`;
        return;
      }
    }
  }

  // Gate 3 — Path-allowlist (D-13)
  if (tool === "openFile" || tool === "saveFileAs") {
    const path = args.path as string | undefined;
    if (!path || !isPathAllowed(path)) {
      sendError(ws, id, {
        code: "path_not_permitted",
        message: "Path is outside the permitted directory.",
        hint: "Open the target directory in the app first, or navigate to the file via the file picker.",
      });
      agentLogger.warn`${tool} path_not_permitted path=${String(path)}`;
      return;
    }
  }

  // Gate 4 — Pre-open auto-flush (D-12)
  if (tool === "openFile") {
    try {
      await flushPending();
    } catch (err) {
      sendError(ws, id, {
        code: "save_error",
        message: `Failed to flush pending edits before openFile: ${String(err)}`,
        hint: "Check disk space and file permissions.",
      });
      agentLogger.error`openFile pre-flush failed: ${String(err)}`;
      return;
    }
  }

  // Forward to the renderer dispatcher
  try {
    const rpc = mainWindow.webview.rpc;
    if (!rpc) {
      sendError(ws, id, {
        code: "internal_error",
        message: "Renderer RPC not yet available.",
        hint: "Try again after the window is shown.",
      });
      return;
    }
    const result = await rpc.request.agentRequest({ tool, args });

    // Post-success: register saveFileAs path with the allowlist (D-13)
    if (
      tool === "saveFileAs" &&
      result.ok &&
      typeof (result.data as { filePath?: unknown })?.filePath === "string"
    ) {
      pushDialogAllowlistPath((result.data as { filePath: string }).filePath);
    }

    const durationMs = Date.now() - start;
    if (result.ok) {
      sendOk(ws, id, result.data);
      agentLogger.info`Agent tool ok tool=${tool} durationMs=${durationMs}`;
    } else {
      sendError(ws, id, {
        code: result.code,
        message: result.error,
        hint: result.hint,
        data: result.data,
      });
      agentLogger.warn`Agent tool err tool=${tool} code=${result.code} durationMs=${durationMs}`;

    });

  }
}

  // re-check on the actual write path.

}

}

}
```
