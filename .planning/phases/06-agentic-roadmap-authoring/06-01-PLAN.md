    ```typescript
    import { readSentinel } from "../sentinel";

    type WsClientLike = {
      request<T = unknown>(method: string, params: Record<string, unknown>): Promise<T>;
    };

    type McpResult = {
      content: Array<{ type: "text"; text: string }>;
      isError?: boolean;
    };

    /**
     * Shared MCP registerTool callback. Phase 6 anti-sprawl: 17 net-new server.registerTool
     * calls in plugins/claude-code/src/server.ts (Plan 06-05) all delegate here. This helper
     * is the SINGLE place we test the success/error formatting (test 5 in agent-contracts.test.ts);
     * per-tool unit tests are not needed because the per-tool delta is just the tool name string.
     */
    export function agentToolCallback(
      method: string,
      wsClient: WsClientLike,
    ): (args: Record<string, unknown>) => Promise<McpResult> {
      return async (args) => {
        try {
          const result = await wsClient.request(method, args ?? {});
          return {
            content: [{ type: "text", text: JSON.stringify(result, null, 2) }],
          };
        } catch (err: unknown) {
          // Distinguish: sentinel missing (app off) vs sentinel present + structured error.
          const sentinel = await readSentinel();
          if (!sentinel.ok) {
            return {
              content: [
                {
                  type: "text",
                  text: "Roadmap Viewer is not running. Start the app and retry.",
                },
              ],
              isError: true,
            };
          }
          const e = err as { code?: string; hint?: string; message?: string };
          const code = e.code ?? "internal_error";
          const message = e.message ?? "Unknown error";
          const hint = e.hint ? ` ${e.hint}` : "";
          return {
            content: [{ type: "text", text: `Error (${code}): ${message}${hint}` }],
            isError: true,
          };
        }
      };
    }
    ```
