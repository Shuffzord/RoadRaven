```typescript
// Phase 6 / PLUG-AGENT-* — renderer-side dispatcher for the agentRequest RPC.
// One switch on tool string → existing roadmapStore action. Drawer audit
// events fired AFTER successful mutations (D-09 / SAFETY-02). Read tools
// never write drawer events. Cycle detection (moveNode), node-not-found,
// and cascade gates live here because they need the live schema.
//
// Bun-side handler (../bun/agentRequestHandler.ts) owns the kill-switch,
// cross-ref ownership boundary, and path-allowlist gates BEFORE this dispatcher
// is called.

import type {
  RoadmapNode,
  RoadmapSchema,
} from "../../../../../packages/core/src/schema";
import type { IntegrationEvent } from "../../../../../shared/types";

type AgentResult =
  | { ok: true; data: unknown }
  | { ok: false; error: string; code: string; hint?: string; data?: unknown };

export async function handleAgentRequest(
  tool: string,
  args: Record<string, unknown>,
): Promise<AgentResult> {
  const { useRoadmapStore } = await import("../store/roadmapStore");
  const { useEventLogStore } = await import("../store/eventLogStore");
  const store = useRoadmapStore.getState();
  const drawer = useEventLogStore.getState();

  // Special-case tools that don't require a loaded schema
  if (tool === "getOpenFile") {
    return {
      ok: true,
      data: {
        filePath: store.filePath,
        isUntitled: store.isUntitled,
        title: store.schema?.title ?? null,
        nodeCount: store.nodeIndex.size,
      },
    };
  }

  if (tool === "createRoadmap") {
    // D-01: mirror File > New (newUntitledSchema). Optional title/configs are
    // applied to the resulting in-memory schema. Schema is unchanged on disk
    // until the agent calls saveFileAs.
    store.newUntitledSchema();
    const fresh = useRoadmapStore.getState().schema;
    if (!fresh) {
      return { ok: false, error: "Failed to initialise untitled schema.", code: "internal_error" };
    }
    if (typeof args.title === "string" && args.title.length > 0) {
      fresh.title = args.title;
    }
    if (Array.isArray(args.statusConfig)) {
      fresh.statusConfig = args.statusConfig as RoadmapSchema["statusConfig"];
    }
    if (Array.isArray(args.typeConfig)) {
      fresh.typeConfig = args.typeConfig as RoadmapSchema["typeConfig"];
    }
    appendAgentDrawerEvent(drawer, {
      tool,
      nodeId: "__lifecycle__",
      status: "unknown",
      label: `Claude → createRoadmap "${fresh.title}"`,
      args,
    });
    return { ok: true, data: { schema: fresh, isUntitled: true } };
  }

  if (tool === "openFile") {
    // D-12 / FILE-03: Bun side already flushed pending autosave + checked
    // allowlist. We just call loadFile RPC. electroview is read lazily.
    const { electroview } = await import("../rpc");
    if (!electroview?.rpc) {
      return { ok: false, error: "Renderer not ready.", code: "internal_error" };
    }
    const path = args.path as string;
    const res = await electroview.rpc.request.loadFile({ path });
    if (!res.data) {
      const detail = res.errors?.[0]?.message ?? "load failed";
      return { ok: false, error: `Failed to read file: ${detail}`, code: "file_read_error" };
    }
    appendAgentDrawerEvent(drawer, {
      tool,
      nodeId: "__lifecycle__",
      status: "unknown",
      label: "Claude → openFile",
      args,
    });
    return { ok: true, data: { schema: res.data, filePath: path } };
  }

  // Every remaining tool requires a loaded schema (D-06)
  const schema = store.schema;
  if (!schema) {
    return {
      ok: false,
      error: "No roadmap file is loaded.",
      code: "no_file_loaded",
      hint: "Open a roadmap or call openFile(path).",
    };
  }

  switch (tool) {
    // -- Read tools (no drawer events) --
    case "getRoadmap":
      return {
        ok: true,
        data: { schema, filePath: store.filePath, isUntitled: store.isUntitled },
      };

    case "getNode": {
      const nodeId = args.nodeId as string;
      const node = store.nodeIndex.get(nodeId);
      if (!node) return errNotFound(nodeId);
      const ancestry = buildAncestry(schema.nodes, nodeId);
      return { ok: true, data: { node, parentId: ancestry.parentId, ancestorIds: ancestry.ancestorIds } };
    }

    case "findNodes": {
      const filter = args as FindNodesFilter;
      const matches = runFindNodesFilter(schema.nodes, filter);
      return { ok: true, data: { nodes: matches } };
    }

    case "getStatusConfig":
      return { ok: true, data: { statusConfig: schema.statusConfig ?? [] } };

    case "getTypeConfig":
      return { ok: true, data: { typeConfig: schema.typeConfig ?? [] } };

    // -- Create tools --
    case "createNode": {
      const parentId = args.parentId as string;
      if (!store.nodeIndex.get(parentId)) {
        return { ok: false, error: `Parent node '${parentId}' not found.`, code: "node_not_found" };
      }
      const newId = store.addChild(parentId, args.title as string);
      if (!newId) {
        return { ok: false, error: `Parent node '${parentId}' not found.`, code: "node_not_found" };
      }
      // Apply optional fields via the in-place store actions
      if (typeof args.status === "string") store.updateNodeStatus(newId, args.status);
      if (typeof args.type === "string") store.updateNodeType(newId, args.type);
      if (typeof args.notes === "string") store.updateNodeNotes(newId, args.notes);
      if (args.metadata && typeof args.metadata === "object") {
        store.updateNodeMetadata(newId, args.metadata as Record<string, unknown>);
      }
      const created = useRoadmapStore.getState().nodeIndex.get(newId);
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId: newId,
        status: created?.status ?? "unknown",
        label: `Claude → createNode "${created?.title ?? args.title}"`,
        args,
      });
      return { ok: true, data: { nodeId: newId } };
    }

    // -- Update tools --
    case "renameNode": {
      const nodeId = args.nodeId as string;
      const node = store.nodeIndex.get(nodeId);
      if (!node) return errNotFound(nodeId);
      const oldTitle = node.title;
      store.renameNode(nodeId, args.title as string);
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId,
        status: node.status,
        label: `Claude → renamed "${oldTitle}" → "${args.title}"`,
        args,
      });
      return { ok: true, data: { ok: true } };
    }

    case "updateNodeType": {
      const nodeId = args.nodeId as string;
      const node = store.nodeIndex.get(nodeId);
      if (!node) return errNotFound(nodeId);
      store.updateNodeType(nodeId, args.type as string);
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId,
        status: node.status,
        label: `Claude → type "${node.title}" → ${args.type}`,
        args,
      });
      return { ok: true, data: { ok: true } };
    }

    case "updateNodeNotes": {
      const nodeId = args.nodeId as string;
      const node = store.nodeIndex.get(nodeId);
      if (!node) return errNotFound(nodeId);
      store.updateNodeNotes(nodeId, args.notes as string);
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId,
        status: node.status,
        label: `Claude → updated notes on "${node.title}"`,
        args,
      });
      return { ok: true, data: { ok: true } };
    }

    case "updateNodeMetadata": {
      const nodeId = args.nodeId as string;
      const node = store.nodeIndex.get(nodeId);
      if (!node) return errNotFound(nodeId);
      // D-04 PATCH semantics — applied here, NOT in the store action (RESEARCH L-07)
      const patch = args.patch as Record<string, unknown | null>;
      const current = node.metadata ?? {};
      const next: Record<string, unknown> = { ...current };
      const changedKeys: string[] = [];
      for (const [k, v] of Object.entries(patch)) {
        changedKeys.push(k);
        if (v === null) delete next[k];
        else next[k] = v;
      }
      store.updateNodeMetadata(nodeId, next);
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId,
        status: node.status,
        label: `Claude → metadata "${node.title}" (${changedKeys.join(", ")})`,
        args,
      });
      return { ok: true, data: { ok: true, metadata: next } };
    }

    case "moveNode": {
      const nodeId = args.nodeId as string;
      const newParentId = args.newParentId as string;
      const position = typeof args.position === "number" ? args.position : undefined;
      const node = store.nodeIndex.get(nodeId);
      if (!node) return errNotFound(nodeId);
      if (!store.nodeIndex.get(newParentId)) {
        return { ok: false, error: `Parent node '${newParentId}' not found.`, code: "node_not_found" };
      }
      // Cycle check: newParentId must NOT be in the subtree rooted at nodeId.
      if (isDescendantOf(node, newParentId)) {
        return {
          ok: false,
          error: "Cannot move a node into its own subtree.",
          code: "move_would_create_cycle",
        };
      }
      store.moveNode(nodeId, newParentId, position);
      const newParent = useRoadmapStore.getState().nodeIndex.get(newParentId);
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId,
        status: node.status,
        label: `Claude → moved "${node.title}" to under "${newParent?.title ?? newParentId}"`,
        args,
      });
      return { ok: true, data: { ok: true } };
    }

    case "updateNodeStatus": {
      // Phase 4 carry-forward (PLUG-AGENT-UPDATE-06): now also writes drawer event.
      const nodeId = args.nodeId as string;
      const node = store.nodeIndex.get(nodeId);
      if (!node) return errNotFound(nodeId);
      store.updateNodeStatus(nodeId, args.status as string);
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId,
        status: args.status as string,
        label: `Claude → status "${node.title}" → ${args.status}`,
        args,
      });
      return { ok: true, data: { ok: true } };
    }

    // -- Delete tool --
    case "deleteNode": {
      const nodeId = args.nodeId as string;
      const node = store.nodeIndex.get(nodeId);
      if (!node) return errNotFound(nodeId);
      const childCount = node.children?.length ?? 0;
      const isRoot = !findParentInTree(schema.nodes, nodeId);
      const isLastRoot = isRoot && schema.nodes.length === 1;
      if (isLastRoot) {
        return {
          ok: false,
          error: "Cannot delete the last root node.",
          code: "cannot_delete_last_root",
          hint: "Add a sibling root node first, or close the file.",
        };
      }
      if (childCount > 0 && args.cascade !== true) {
        return {
          ok: false,
          error: `Node has ${childCount} children. Pass cascade:true to delete the subtree.`,
          code: "cascade_required",
          data: { childCount },
        };
      }
      const result = store.deleteNode(nodeId);
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId,
        status: node.status,
        label: `Claude → deleted "${node.title}" (${result.deletedCount} nodes)`,
        args,
      });
      return { ok: true, data: { ok: true, deletedCount: result.deletedCount } };
    }

    // -- File-lifecycle tools --
    case "saveFile": {
      const { electroview } = await import("../rpc");
      if (!electroview?.rpc) {
        return { ok: false, error: "Renderer not ready.", code: "internal_error" };
      }
      const res = await electroview.rpc.request.saveFile({ schema });
      if (!res.ok) {
        return { ok: false, error: res.error ?? "save failed", code: "save_error" };
      }
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId: "__lifecycle__",
        status: "unknown",
        label: "Claude → saveFile",
        args,
      });
      return { ok: true, data: { ok: true } };
    }

    case "saveFileAs": {
      // Bun gate (Plan 02) already verified the path. We bypass the native-dialog
      // saveFileAs RPC and use saveFile with explicit filePath; saveFileHandler
      // does its own allowlist check (defense in depth).
      const { electroview } = await import("../rpc");
      if (!electroview?.rpc) {
        return { ok: false, error: "Renderer not ready.", code: "internal_error" };
      }
      const filePath = args.path as string;
      const res = await electroview.rpc.request.saveFile({ schema, filePath });
      if (!res.ok) {
        return { ok: false, error: res.error ?? "save failed", code: "save_error" };
      }
      appendAgentDrawerEvent(drawer, {
        tool,
        nodeId: "__lifecycle__",
        status: "unknown",
        label: "Claude → saveFileAs",
        args,
      });
      return { ok: true, data: { ok: true, filePath } };
    }

    default:
      return {
        ok: false,
        error: `Unknown agent tool: '${tool}'.`,
        code: "unknown_tool",
        hint: "Update the plugin to a version that matches the app.",
      };
  }

function appendAgentDrawerEvent(
  drawer: { appendEvents: (events: IntegrationEvent[]) => void },
  input: DrawerEventInput,

    status: input.status,

    },
  };

}

function errNotFound(nodeId: string): AgentResult {
  return {
    ok: false,
    error: `Node '${nodeId}' not found in the loaded roadmap.`,
    code: "node_not_found",
    hint: "Call getRoadmap or findNodes to discover node IDs.",
  };
}

function findParentInTree(nodes: RoadmapNode[], nodeId: string): RoadmapNode | null {
  for (const n of nodes) {
    if (n.children) {
      for (const c of n.children) {
        if (c.id === nodeId) return n;
      }
      const deeper = findParentInTree(n.children, nodeId);
      if (deeper) return deeper;
    }
  }
  return null;
}

function buildAncestry(
  nodes: RoadmapNode[],
  nodeId: string,
): { parentId: string | null; ancestorIds: string[] } {
  const ancestors: string[] = [];
  let cursor = nodeId;
  // Walk up until we hit a root (no parent)
  while (true) {
    const parent = findParentInTree(nodes, cursor);
    if (!parent) break;
    ancestors.unshift(parent.id);
    cursor = parent.id;
  }
  return {
    parentId: ancestors.length > 0 ? ancestors[ancestors.length - 1] : null,
    ancestorIds: ancestors,
  };
}

function isDescendantOf(rootNode: RoadmapNode, candidateId: string): boolean {
  if (rootNode.id === candidateId) return true;
  if (!rootNode.children) return false;
  for (const c of rootNode.children) {
    if (isDescendantOf(c, candidateId)) return true;
  }
  return false;
}

function runFindNodesFilter(
  nodes: RoadmapNode[],
  filter: FindNodesFilter,
): Array<{ node: RoadmapNode; parentId: string | null }> {
  const out: Array<{ node: RoadmapNode; parentId: string | null }> = [];
  const titleNeedle = filter.titleContains?.toLowerCase();
  const wantParentId = filter.parentId;

  function walk(arr: RoadmapNode[], parent: RoadmapNode | null): void {
    for (const n of arr) {
      const matchesTitle = titleNeedle === undefined || n.title.toLowerCase().includes(titleNeedle);
      const matchesStatus = filter.status === undefined || n.status === filter.status;
      const matchesType = filter.type === undefined || n.type === filter.type;
      const matchesMeta = (() => {
        if (filter.metaKey === undefined) return true;
        const meta = n.metadata ?? {};
        if (!(filter.metaKey in meta)) return false;
        if (filter.metaValue === undefined) return true;
        return JSON.stringify(meta[filter.metaKey]) === JSON.stringify(filter.metaValue);
      })();
      const matchesParent = wantParentId === undefined || (parent !== null && parent.id === wantParentId);
      if (matchesTitle && matchesStatus && matchesType && matchesMeta && matchesParent) {
        out.push({ node: n, parentId: parent?.id ?? null });
      }
      if (n.children && n.children.length > 0) walk(n.children, n);
    }
  }
  walk(nodes, null);
  return out;
}
```
