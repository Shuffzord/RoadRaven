function bridgeWs(clientWs: MockWebSocket): void {
  const original = clientWs.send;
  clientWs.send = vi.fn(async (raw: string) => {
    original.call(clientWs, raw);
    const frame = JSON.parse(raw);
    if (frame.type !== "request") return;
    const fakeServerWs = {
      data: { id: "test", source: "claude-code" },
      send: (responseRaw: string) => {
        // Feed response back to the wsClient's onmessage listener
        clientWs.emit("message", { data: responseRaw });

          request: {
            agentRequest: async ({ tool, args }: { tool: string; args: Record<string, unknown> }) =>

        },
