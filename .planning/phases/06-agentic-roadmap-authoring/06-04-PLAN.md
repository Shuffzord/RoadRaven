```typescript
moveNode: (nodeId, newParentId, position) => {
  const schema = get().schema;
  if (!schema) return;
  const nodes = schema.nodes;

  const found = findParentAndIndex(nodes, nodeId);
  if (!found) return;
  const node = found.parentArray[found.index];

  // Verify newParentId exists (for non-root moves; null/empty newParentId not supported in v1).
  if (!get().nodeIndex.get(newParentId)) return;

  const currentParentId = found.parent ? found.parent.id : null;

  // Remove from current parent
  const nodesAfterRemove = immutablyReplaceArray(nodes, currentParentId, (arr) => {
    const copy = [...arr];
    copy.splice(found.index, 1);
    return copy;
  });

  // Insert at new parent
  const nextNodes = immutablyReplaceArray(nodesAfterRemove, newParentId, (arr) => {
    const copy = [...arr];
    const pos = position !== undefined ? Math.min(position, copy.length) : copy.length;
    copy.splice(pos, 0, node);
    return copy;
  });

  bumpStructural(nextNodes);
},
```

**Cycle detection — done in agentRpcHandler BEFORE calling store.moveNode:**
```typescript
// Walk descendants of nodeId; if newParentId is found, it's a cycle.
function isDescendantOf(rootNodeId: string, candidateId: string, nodeIndex: Map<string, RoadmapNode>): boolean {
  const root = nodeIndex.get(rootNodeId);
  if (!root) return false;
  const stack = [...(root.children ?? [])];
  while (stack.length) {
    const n = stack.pop()!;
    if (n.id === candidateId) return true;
    if (n.children) stack.push(...n.children);
  }
  return false;
}
```

    function isDescendantOf(
      rootNodeId: string,
      candidateId: string,
      nodeIndex: Map<string, RoadmapNode>,
    ): boolean {
      const root = nodeIndex.get(rootNodeId);
      if (!root) return false;
      const stack: RoadmapNode[] = [...(root.children ?? [])];
      while (stack.length) {
        const n = stack.pop()!;
        if (n.id === candidateId) return true;
        if (n.children) stack.push(...n.children);
      }
      return false;
