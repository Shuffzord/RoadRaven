    EDIT 2 — Add a module-level helper function AFTER the `const server = new McpServer({...})` block (current line 18) and BEFORE the first `server.registerTool(...)` call. This eliminates the boilerplate every new tool would otherwise duplicate:

```typescript
/**
 * Phase 6 — shared callback wrapper for every agent tool that flows through
 * wsClient.request. Returns the MCP tool-callback shape:
 *  - On success: { content: [{ type: "text", text: <pretty-printed result> }] }
 *  - On structured error from Bun: { content: [...], isError: true } with the
 *    error code embedded in the text (MCP has no machine-readable error field
 *    for tool results — RESEARCH Landmine L-09).
 *  - On sentinel-missing: distinguishes "Roadmap Viewer is not running" from
 *    "running but unreachable" — same UX as the existing updateNodeStatus
 *    sentinel-missing branch (server.ts pre-Phase-6).
 */
async function agentToolCallback(
  method: string,
  args: Record<string, unknown>,
): Promise<{ content: Array<{ type: "text"; text: string }>; isError?: boolean }> {
  try {
    const result = await wsClient.request<unknown>(method, args);
    // Success path — pretty-print the response. The renderer dispatcher
    // returns the `data` payload (already unwrapped from the {ok:true,data}
    // discriminated union by wsClient.request's resolve path).
    return {
      content: [{ type: "text", text: JSON.stringify(result, null, 2) }],
    };
  } catch (err: unknown) {
    // First check sentinel state to distinguish "app not running" from a real
    // structured error (matches the existing updateNodeStatus error handling).
    const sentinel = await readSentinel();
    if (!sentinel.ok) {
      return {
        content: [
          {
            type: "text",
            text: "Roadmap Viewer is not running. Start the app and retry.",
          },
        ],
        isError: true,
      };
    }
    // Pull structured error fields off the Error instance (wsClient.ts
    // attaches code/hint/data via Object.assign on the rejection).
    const code = (err as { code?: string }).code ?? "internal_error";
    const message = (err as Error).message ?? "Unknown error";
    const hint = (err as { hint?: string }).hint ?? "";
    return {
      content: [
        {
          type: "text",
          text: `Error (${code}): ${message}${hint ? ` ${hint}` : ""}`,
        },
      ],
      isError: true,
    };
  }
}
```
